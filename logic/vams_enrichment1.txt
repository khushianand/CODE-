"""Dedicated VAMS enrichment engine.

This module maps VAMS rows into:
- New Data: fill-only VAMS-managed fields after key match (no overwrite)
- Unique Data: merge-only VAMS-managed fields (no overwrite)
"""

from __future__ import annotations

import re
import pandas as pd

from logic.parser import merge_semicolon, split_values

VAMS_COLUMNS = [
    "Release Remediation Plan",
    "Release Remediation Date",
    "Expert Severity",
    "Expert Score",
    "Remediation Reference ID",
    "Disposition",
    "VAMS (PSL comments)",
    "MSS Comments",
]


def _norm(value: object) -> str:
    return str(value).strip().lower()

def _norm_port(value: object) -> str:
    token = _norm(value)
    if re.match(r"^\d+\.0+$", token):
        return token.split(".", 1)[0]
    return token

def _fallback_cve_tokens(row: pd.Series) -> list[str]:
    raw_cve_tokens = split_values(row.get("CVE", ""))
    normalized: list[str] = []
    for token in raw_cve_tokens:
        t = str(token).strip()
        if not t:
            continue
        if re.match(r"(?i)^cve-\d{4}-\d+$", t):
            #regex code
            normalized.append(t)
            continue
        # Non-CVE strings (e.g. NS-123456) are treated as id-like and reduced
        # to exact numeric entities for strict fallback matching.
        normalized.extend(re.findall(r"\d+", t))
    if normalized:
        return normalized
    # Strict fallback: match only exact numeric entities from plugin/scanner id.
    plugin_tokens = re.findall(r"\d+", str(row.get("Plugin ID", "")))
    return plugin_tokens



def _extract_port_tokens(value: object) -> list[str]:
    """Extract port tokens, including cases where VAMS embeds ports in '(...)' text."""
    text = str(value or "")
    bracket_tokens = re.findall(r"\(([^)]*)\)", text)
    parsed = split_values(";".join(bracket_tokens) if bracket_tokens else text)
    return parsed or ([text.strip()] if text.strip() else [])


def _extract_host_and_embedded_ports(value: object) -> tuple[str, list[str]]:
    """Parse VAMS Host field like 'ww.xx.yy.zz (443)' -> ('ww.xx.yy.zz', ['443'])."""
    text = str(value or "").strip()
    if not text:
        return "", []
    ip_or_host = re.sub(r"\s*\([^)]*\)\s*$", "", text).strip()
    embedded_ports = re.findall(r"\(([^)]*)\)", text)
    ports = split_values(";".join(embedded_ports)) if embedded_ports else []
    return ip_or_host, ports

def _row_keys(row: pd.Series, include_port: bool) -> set[str]:
    name = _norm(row.get("Name", ""))

    # Supports both:
    # - "Host / Image"
    # - "Host"
    #host_value = row.get("Host / Image", row.get("Host", ""))
    host_value = row.get("Host / Image", "")

    if not str(host_value).strip() or str(host_value).lower() == "nan":
        host_value = row.get("Host", "")

    parsed_host, embedded_ports = _extract_host_and_embedded_ports(host_value)

    hosts = (
        split_values(parsed_host)
        or ([parsed_host] if parsed_host else split_values(host_value))
        or [host_value]
    )

    cves = _fallback_cve_tokens(row) or [row.get("CVE", "")]

    if include_port:
        ports = (
            _extract_port_tokens(row.get("Port", ""))
            or embedded_ports
            or [row.get("Port", "")]
        )

        return {
            f"{name}|{_norm(h)}|{_norm_port(p)}|{_norm(c)}"
            for h in hosts
            for p in ports
            for c in cves
        }

    return {
        f"{name}|{_norm(h)}|{_norm(c)}"
        for h in hosts
        for c in cves
    }

def _value_in_merged_cell(target: str, merged_cell: str) -> bool:
    #return _norm(target) in {_norm(token) for token in split_values(merged_cell)}
    return _norm_port(target) in {_norm_port(token) for token in split_values(merged_cell)}


def enrich_with_vams(new_df: pd.DataFrame, unique_df: pd.DataFrame, vams_df: pd.DataFrame, *, new_data_match_include_port: bool = True) -> tuple[pd.DataFrame, pd.DataFrame]:
    """Apply VAMS fields into output dataframes with robust matching."""
    if vams_df.empty:
        return new_df, unique_df
    out_new, out_unique = new_df.copy(), unique_df.copy()
    index: dict[str, pd.Series] = {}
    for _, row in vams_df.iterrows():
        for key in _row_keys(row, include_port=new_data_match_include_port):
            index[key] = row

    for idx, row in out_new.iterrows():
        match = next((index.get(k) for k in _row_keys(row, include_port=new_data_match_include_port) if index.get(k) is not None), None)
        if match is not None:
            for col in VAMS_COLUMNS:
                #out_new.at[idx, col] = str(match.get(col, "")).strip()
                existing = out_new.at[idx, col] if col in out_new.columns else ""
                incoming = str(match.get(col, "")).strip()
                if not str(existing).strip() and incoming:
                    out_new.at[idx, col] = incoming


    for uidx, urow in out_unique.iterrows():
        for _, vrow in vams_df.iterrows():
            if str(urow["Name"]).strip() != str(vrow["Name"]).strip():
                continue

            #vams_host, embedded_ports = _extract_host_and_embedded_ports(vrow["Host / Image"])
            #host_value = vrow.get("Host / Image", vrow.get("Host", ""))
            host_value = vrow.get("Host / Image", "")

            if not str(host_value).strip() or str(host_value).lower() == "nan":
                host_value = vrow.get("Host", "")
            vams_host, embedded_ports = _extract_host_and_embedded_ports(host_value)
            vams_port_for_match = vrow["Port"] if str(vrow.get("Port", "")).strip() else "; ".join(embedded_ports)
            urow_cves = _fallback_cve_tokens(urow)
            vrow_cves = _fallback_cve_tokens(vrow)
            cve_match_value = "; ".join(vrow_cves) if vrow_cves else vrow["CVE"]
            
            if not (
                _value_in_merged_cell(vams_host, urow["Host / Image"])
                and _value_in_merged_cell(vams_port_for_match, urow["Port"])
                #and _value_in_merged_cell(vrow["CVE"], urow["CVE"])
               and (
                    any(_value_in_merged_cell(token, "; ".join(urow_cves) if urow_cves else urow["CVE"]) for token in vrow_cves)
                    or _value_in_merged_cell(cve_match_value, urow["CVE"])
                ) 
            ):
                continue
            for col in VAMS_COLUMNS:
                out_unique.at[uidx, col] = merge_semicolon([out_unique.at[uidx, col], vrow.get(col, "")])

    return out_new, out_unique
