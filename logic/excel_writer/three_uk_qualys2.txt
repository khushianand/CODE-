def normalized_name(value: object) -> str:
#    return " ".join(str(value).strip().lower().replace("_", " ").split())
    if value is None:
        return ""

    text = str(value).replace("\ufeff", "").strip()
    if text.casefold() in {"", "nan", "nat", "none", "<na>"}:
        return ""

    return " ".join(text.lower().replace("_", " ").split())


def column_values(df: pd.DataFrame, column: object) -> pd.Series:
    values = df.loc[:, column]
    if isinstance(values, pd.Series):
        return values.fillna("").astype(str).str.strip()

    result = pd.Series([""] * len(df), index=df.index, dtype="object")
    for position in range(values.shape[1]):
        series = values.iloc[:, position].fillna("").astype(str).str.strip()
        empty_rows = result.eq("")
        result.loc[empty_rows] = series.loc[empty_rows]

    return result

def fill_empty_values(result: pd.Series, values: pd.Series) -> pd.Series:
    empty_rows = result.eq("") & values.ne("")
    result.loc[empty_rows] = values.loc[empty_rows]
    return result

def source_series(df: pd.DataFrame, *source_columns: str) -> pd.Series:
    normalized_columns: dict[str, list[object]] = {}
    for column in df.columns:
        normalized = normalized_name(column)
        if normalized:
            normalized_columns.setdefault(normalized, []).append(column)

    result = pd.Series([""] * len(df), index=df.index, dtype="object")
#    normalized_columns = {normalized_name(col): col for col in df.columns}

    for source_col in source_columns:
        for actual_col in normalized_columns.get(normalized_name(source_col), []):
            if actual_col in df.columns:
                result = fill_empty_values(result, column_values(df, actual_col))

    return result


_SEVERITY_LABELS = {
    "low": "LOW",
    "medium": "MEDIUM",
    "med": "MEDIUM",
    "moderate": "MEDIUM",
    "high": "HIGH",
    "critical": "CRITICAL",
    "crit": "CRITICAL",
}

def _severity_score(value: object) -> float | None:
    if pd.isna(value):
        return None

    text = str(value).strip()
    if not text:
        return None

    label = _SEVERITY_LABELS.get(normalized_name(text))
    if label is not None:
        return None

    match = re.search(r"\d+(?:[.,]\d+)?", text)
    if match is None:
        return None

    try:
        return float(match.group(0).replace(",", "."))
    except ValueError:
        return None


def is_three_uk_qualys_total_sheet(project: str, scanner: str, sheet_name: str) -> bool:
    """Return True only for 3UK + Qualys Total Data / Total Vulnerability output sheets."""
    return (
        str(project).strip().casefold() == THREE_UK_PROJECT
        and str(scanner).strip().casefold() == QUALYS_SCANNER
        and str(sheet_name).strip().casefold() in THREE_UK_QUALYS_TOTAL_SHEET_NAMES
    )


def criticality_from_severity(value: object) -> str:
    label = _SEVERITY_LABELS.get(normalized_name(value))
    if label is not None:
        return label

    severity = _severity_score(value)
    if severity is None:
        return ""

    if severity <= 4.0:
        return "LOW"
    if 4.1 < severity <= 7.0:
        return "MEDIUM"
    if 7.1 < severity <= 8.5:
        return "HIGH"
    if 8.5 < severity <= 10.0:
        return "CRITICAL"
    return ""


def criticality_series(df: pd.DataFrame) -> pd.Series:
    current_criticality = source_series(df, "Criticality").apply(criticality_from_severity)
    severity = source_series(
        df,
        "Severity",
    
    ).apply(criticality_from_severity)

    return current_criticality.where(current_criticality.ne(""), severity)
